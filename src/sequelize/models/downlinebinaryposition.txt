'use strict';
module.exports = (sequelize, DataTypes) => {
  const DownLineBinaryPosition = sequelize.define('DownLineBinaryPosition', {
    id: {
      primaryKey: true,
      type: DataTypes.UUID,
      defaultValue: DataTypes.UUIDV4
    },
    distributorUsername: {
      type: DataTypes.STRING
    },
    position: {
      type: DataTypes.ENUM,
      values: ['left', 'right']
    },
    positionNumber: {
      type: DataTypes.INTEGER
    },
    createdAt: {
      allowNull: false,
      type: DataTypes.BIGINT,
      defaultValue: new Date().getTime()
    },
    updatedAt: {
      allowNull: false,
      type: DataTypes.BIGINT,
      defaultValue: new Date().getTime()
    }
  }, {});
  DownLineBinaryPosition.associate = function(models) {
    // associations can be defined here
    DownLineBinaryPosition.belongsTo(models.Distributor, {
      as: 'distributor',
      foreignKey: 'distributorUsername'
    });
  };
  return DownLineBinaryPosition;
};
